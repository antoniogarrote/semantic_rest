  \EjBodyH=#3

    \begin{minipage}[t]{#2}

{
    \end{minipage}
  \end{lrbox}
  \EjTotalH=\EjBodyH
  \advance\EjTotalH \EjGap
  \EjHeaderB=0.5\EjTotalH
  \advance\EjHeaderB -0.25cm
  \EjFooterB=-0.5\EjTotalH
  \advance\EjFooterB -0.25cm
